Vim基本配置

Vim为不同的用户做不同的配置。主要的配置包括两个方面：配置项+插件
配置项：.vimrc 文件 参考文档：http://www.cnblogs.com/chijianqiang/archive/2012/10/31/2748729.html

操作：

  • 打开Terminal
  • 键入指令:cd ~
  • 键入指令：vim .vimrc,贴入下面的配置信息,:wq 保存并退出

"语法支持
syn on
"通用配置

"common conf{{
"自动缩进
set ai
"在insert模式下用退格键删除
set bs=2
"代码匹配
set showmatch
"总是显示状态航
set laststatus=2
"以下三个配置配合使用，设置tab和缩进空格数
set expandtab
set shiftwidth=4
set tabstop=4
"为光标所在行加下划线
set cursorline
"显示行号
set number
"文件在vim之外修改过，自动重新读入
set autoread

"检索时忽略大小
set ignorecase
"使用utf-8或gbk或IOS-8859-1打开文件
set fileencodings=utf-8,gbk,IOS-8859-1
"检索时高亮显示匹配项
set hls
"采用中文帮助系统
set helplang=cn
"代码折叠
set foldmethod=syntax
"}}end of common conf

"conf for tabs, 为标签页进行的配置，通过ctrl h/l切换标签等
let mapleader = ","
nnoremap <C-l> gt
nnoremap <C-h> gT
nnoremap <leader>t : table<CR>

"conf for plugins {{ 插件相关的配置
"状态栏的配置
"powerline{
set guifont=PowerlineSymbols\ for\ Powerline
set nocompatible
set t_Co=256
let g:Powerline_symbols = 'fancy'
"}
"pathogen是Vim用来管理插件的插件
"pathogen{
"call pathogen#infect()
"}
"}}
